class GameCharacter:
    def __init__(self, name):
        #Name is the string given at the moment of the instantiation.
        self._name = name
        
        self._health = 100
        self._mana = 50
        self._level = 1

    #A getter
    @property #sets it as read only
    def name(self):
        return self._name #Tells us what the name is

    @property
    def health(self):
        return self._health
    
    #Sets the value of health and creates its max and mins
    @health.setter
    def health(self, health):
        if health < 0:
            self._health = 0
        elif health > 100:
            self._health = 100
        else:
            self._health = health

    @property
    def mana(self):
        return self._mana

    #Manages MANA levels
    @mana.setter
    def mana(self, mana):
        if mana < 0:
            self._mana = 0
        elif mana > 50:
            self._mana = 50
        else:
            self._mana = mana
    @property
    def level(self):
        return self._level
    
    #Will adjust variables accordingly when a player levels up
    def level_up(self):
        self._level += 1
        self.health = 100
        self.mana = 50
        print(f"{self.name} leveled up to {self.level}!")

    def __str__(self):
        return(f"Name: {self.name}\nLevel: {self.level}\nHealth: {self.health}\nMana: {self.mana}")

Kratos = GameCharacter("Kratos")
print(Kratos)
